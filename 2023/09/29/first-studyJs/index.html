初始代码中包含了我们需要实现的页面数据的数组，我们现在要考虑如何利用这一个数组中的数据完成目标编写
每一页需要显示5个数据条款，在加载的时候还要把题中给出的示例先清除，我们有以下代码实现功能

let currentPage = 1; // 当前页码
let itemsPerPage = 5; // 每页显示的数据条数

// 渲染数据
function renderData() {
  let start = (currentPage - 1) * itemsPerPage;
  let end = start + itemsPerPage;
  let pageData = data.slice(start, end);

  let listGroup = document.querySelector('#list');
  listGroup.innerHTML = ''; // 清空原有数据

  for (let item of pageData) {
    // 创建新的 list-group-item 并添加到 listGroup 中
    let listGroupdiv=document.createElement('div');
    listGroupdiv.className="list-group";
    let listItem = document.createElement('a');
    listItem.href = '#';
    listItem.className = 'list-group-item';
    listItem.innerHTML = `
      <div class="d">
        <h3 class="mb-1">${item.name}</h3>
        <small>${item.price / 100}元</small>
      </div>
      <p class="mb-1">${item.description}</p>
    `;
    listGroupdiv.appendChild(listItem);
    listGroup.appendChild(listGroupdiv);
  }
}

依照示例中的结构，先找到父元素将其内容清空，然后从当前页面找到data数组中这一页开始和结束对应的元素，slice()方法是选择数组中对应的元素，选择后通过innerHTML方法将元素内容全部清空，接下来按照示例创建元素和类名属性，将其加入到父元素中。

下面实现更新页码显示

function updatePagination() 
{
    let pagination = document.getElementById('pagination');
    let totalPages = Math.ceil(data.length / itemsPerPage);
    pagination.textContent = ` 共${totalPages}页，当前${currentPage}页 `;
    }

通过我们上一步设置的页码，将其加入到页脚中，注意取元素的方法${}

接着更新上一页和下一页的状态

// 更新上一页和下一页的状态
function updatePageButtons() {
  let prev = document.getElementById('prev');
  let next = document.getElementById('next');
  let totalPages = Math.ceil(data.length / itemsPerPage);

  prev.className = currentPage === 1 ? 'page-item disabled' : 'page-item';
  next.className = currentPage === totalPages ? 'page-item disabled' : 'page-item';
}

通过判断页面是否是第一页或是最后一页来更改状态

Math.ceil()方法为向上取整

当页面加载时，渲染第一页的数据

window.onload = function() {
  renderData();
  updatePagination();
  updatePageButtons();
};

当点击上一页和下一页时，更新当前页码并重新渲染数据

prev.onclick = function() {
  if (currentPage > 1) {
    currentPage--;
    renderData();
    updatePagination();
    updatePageButtons();
  }
};
next.onclick = function() {
  let totalPages = Math.ceil(data.length / itemsPerPage);
  if (currentPage < totalPages) {
    currentPage++;
    renderData();
    updatePagination();
    updatePageButtons();
  }
};
